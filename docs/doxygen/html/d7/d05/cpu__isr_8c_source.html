292 void page_fault(registers_t regs){
293  uint32_t fault_addr = read_cr2();
294  int present = !(regs.err_code & 0x1); /* Page not present */
295  uint32_t rw = regs.err_code & 0x2; /* Page is read only */
296  uint32_t user = regs.err_code & 0x4; /* User mode */
297  uint32_t reserved = regs.err_code & 0x8; /* Reserved bits is wrote */
298  uint32_t id = regs.err_code & 0x10; /* Instruction fetch */
299  qemu_log("Page fault: ");
300  char* msg = "Переполнение памяти буфера";
301  if (present){
302  qemu_log("NOT PRESENT, ");
303  msg = "Память по данному адресу недоступна";
304  }
305  if (rw){
306  qemu_log("READ ONLY, ");
307  msg = "Память только для чтения";
308  }
309  if (user){
310  qemu_log("USER MODE, ");
311  msg = "Память только для пользователя";
312  }
313  if (reserved){
314  qemu_log("WRITING TO RESERVED BITS, ");
315  msg = "Попытка записи в зарез. биты";
316  }
317  if (id){
318  qemu_log("EIP error ");
319  msg = "Ошибка EIP";
320  }
321  qemu_log("at address (virtual) %x",fault_addr);
322 
323  // Prevent kmallocs (in bsod_screen()) in Page Fault
324  sod_screen_legacy(regs, "CRITICAL_ERROR_PF_PAGE_FAULT", msg, fault_addr);
325 }
