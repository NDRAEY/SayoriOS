199 void createMenuBox(char* title){
200  TUIMode = TUI_MENU_BOX;
201  // Установка отступа экрана в длину (left/right)
202  int padding_w = maxStrLineTUI/4; // 320 - 10 символа; 1024 - 32 символа
203  // Установка отступа экрана в высоту (up)
204  int padding_h = maxHeightLineTUI/4;
205  // Получаем размеры коробки
206  int boxWidth = ww-((padding_w*8)*2);
207  int boxHeight = wh-((padding_h*16));
208  int maxListBox = boxHeight;
209  // Получаем максимальное количество символов на строку в коробке
210  int maxStrLineBox = (boxWidth/8)-4; // 60 - символов при 1024
211 
212  // Высота бокса для меню
213  maxItemScreen = (maxListMenu >10?10:maxListMenu);
214 
215  // Установка максимальных страниц навигаций
216  if (maxListMenu > 0){
217  pageMenuMax = (maxListMenu/maxItemScreen)+1;
218  } else {
219  pageMenuMax = 1;
220  }
221 
222  // Обрезаем заголовок
223  if (strlen(title) > maxStrLineBox){
224  substr(title, title, 0, maxStrLineBox);
225  }
226  // Рисуем бокс и узорчики
227  drawRect(8+(padding_w*8),16*(8+padding_h),boxWidth,16*(6+maxItemScreen),TUI_BASE_COLOR_MAIN);
228  drawRectLine(8+(padding_w*8),16*(8+padding_h),boxWidth,16*(6+maxItemScreen),TUI_TEXT_COLOR_BODY,TUI_BASE_COLOR_MAIN,15);
229 
230  // Отображаем титл
231  setPosX(((3+padding_w)*8));
232  setPosY(16*((10)+padding_h));
233  tty_puts_color(title,TUI_TEXT_COLOR_BODY,TUI_BASE_COLOR_MAIN);
234 
235  // Рисуем пункты меню
236  for (int x = pageMenuCurrent*maxItemScreen; x < maxItemScreen; x++){
237  setPosX(((3+padding_w)*8));
238  setPosY(16*((12+x)+padding_h));
239  if (currentList == x){
240  drawRect(((3+padding_w)*8),16*((12+x)+padding_h),maxStrLineBox*8, 16,TUI_BASE_COLOR_ITEM);

242  // Помечаем что, он является текущим
243  } else {
244  tty_puts_color(listMenu[x],TUI_TEXT_COLOR_BODY,TUI_BASE_COLOR_MAIN);
