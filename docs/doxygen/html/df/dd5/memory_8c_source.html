27 void switch_page_mode(void){
28  uint64_t vaddr = 0;
29  uint32_t frame = 0;
30  physaddr_t paddr = 0;
31  uint32_t table_idx = 0;
32  uint32_t page_idx = 0;
33  uint32_t cr0;
34  uint32_t table_flags = 0;
35  uint32_t page_flags = 0;
36 
37  /* Create kernel page directory */
38  uint32_t* kernel_dir = (uint32_t*) kernel_page_dir;
39  /* Create pointer to page's tables */
40  uint32_t* page_table = (uint32_t*) (kernel_page_dir + PAGE_SIZE);
41 
42  /* Clear kernel directory */
43  memset(kernel_dir, 0, PAGE_SIZE);
44 
45  /* Map all memory pages for kernel */
46  for (vaddr = 0; vaddr < (KERNEL_BASE + KERNEL_SIZE); vaddr += PAGE_SIZE){
47  /* Translate virtual address */
48  frame = vaddr >> PAGE_OFFSET_BITS;
49  table_idx = frame >> PAGE_TABLE_INDEX_BITS;
50  page_idx = frame & PAGE_TABLE_INDEX_MASK;
51 
52  /* Set table flags */
53  table_flags = PAGE_PRESENT | PAGE_WRITEABLE | PAGE_USER;
54 
55  /* Create page table */
56  *(kernel_dir + table_idx) = (uint32_t) (page_table + table_idx*0x400) | table_flags;
57 
58  /* Set page flags */
59  page_flags = PAGE_PRESENT | PAGE_WRITEABLE | PAGE_USER;
60 
61  /* Create page */
62  *(page_table + table_idx*0x400 + page_idx) = paddr | page_flags;
63 
64  paddr += PAGE_SIZE;
65  }
66 
67  /* Load kernel directory address to CR3 */
68  asm volatile ("mov %0, %%cr3"::"r"(kernel_page_dir));
69 
70  /* Switch to paging mode */
71  asm volatile ("mov %%cr0, %0":"=r"(cr0));
72  cr0 |= 0x80000000;
73  asm volatile ("mov %0, %%cr0"::"r"(cr0));
74 }

132  physmemory_pages_block_t* tmp_block;
133  mutex_get(&phys_memory_mutex, true);
134  /* There are no free blocks */
135  if (free_phys_memory_pointer == -1){
136  tmp_block = get_free_block(base);
137 
138  tmp_block->prev = base;
139  tmp_block->next = base;
140  tmp_block->size = count;
141 
142  free_phys_memory_pointer = base;
143  } else {
144  /* Get first free block address */
145  physaddr_t cur_block = free_phys_memory_pointer;
146  do{
147  tmp_block = get_free_block(cur_block);
148 
149  /* If address after current block */
150  if (base == cur_block + (tmp_block->size << PAGE_OFFSET_BITS)){
151  tmp_block->size += count;
152 
153  /* If after new block is other free block*/
154  if (tmp_block->next == base + (count << PAGE_OFFSET_BITS) ){
155  /* Remember next block address*/
156  physaddr_t next_old = tmp_block->next;
157 
158  /* Get next block*/
159  tmp_block = get_free_block(next_old);
160 
161  /* Get new next block address and size */
162  physaddr_t next_new = tmp_block->next;
163  size_t new_count = tmp_block->size;
164 
165  /* Set current block as previous and */
166  /* new next as next */
167  tmp_block = get_free_block(next_new);
168 
169  tmp_block->prev = cur_block;
170 
171  tmp_block = get_free_block(cur_block);
172 
173  tmp_block->next = next_new;
174  tmp_block->size += new_count;
175  }
176 
177  break;
178  }
179 
180  /* If address before current block */
181  if (cur_block == base + (count << PAGE_OFFSET_BITS)){
182  /* Remember old size and next and previous */
183  size_t old_count = tmp_block->size;
184  physaddr_t next = tmp_block->next;
185  physaddr_t prev = tmp_block->prev;
186 
187  /* Get next block */
188  tmp_block = get_free_block(next);
189 
190  /* New block is previous */
191  tmp_block->prev = base;
192 
193  /* Get previous block*/
194  tmp_block = get_free_block(prev);
195 
196  /* New block is next too */
197  tmp_block->next = base;
198 
199  /* Get new block */
200  tmp_block = get_free_block(base);
201 
202  tmp_block->next = next;
203  tmp_block->prev = prev;
204  tmp_block->size += old_count;
205 
206  break;
207  }
208 
209  /* If address between free blocks */
210  if ( cur_block > base){
211  physaddr_t prev = tmp_block->next;
212  tmp_block->prev = base;
213 
214  tmp_block = get_free_block(prev);
215  tmp_block->next = base;
216 
217  tmp_block = get_free_block(base);
218  tmp_block->next = cur_block;
219  tmp_block->prev = prev;
220  tmp_block->size = count;
221 
222  break;
223  }
224 
225  /* or single free block */
226  if (tmp_block->next == free_phys_memory_pointer){
227  tmp_block->prev = base;
228  physaddr_t next = tmp_block->next;
229 
230  tmp_block->next = base;
231 
232  tmp_block = get_free_block(base);
233 
234  tmp_block->prev = cur_block;
235  tmp_block->next = cur_block;
236  tmp_block->size = count;
237 
238  break;
239  }
240 
241  cur_block = tmp_block->next;
242 
243  } while ( cur_block != free_phys_memory_pointer );
244 
245  if (base < free_phys_memory_pointer){
246  free_phys_memory_pointer = base;
247  }
248  }
249  free_pages_count += count;
250  mutex_release(&phys_memory_mutex);

260 physaddr_t alloc_phys_pages(size_t count){
261  physaddr_t result = -1;
262  physmemory_pages_block_t* tmp_block;
263 
264  qemu_log("Allocating %d physical pages", count);
265 
266  mutex_get(&phys_memory_mutex, true);
267 
268  /* Number of free pages less than count - error!*/
269  if (free_pages_count < count)
270  return -1;
271 
272  if (free_phys_memory_pointer != -1){
273  physaddr_t cur_block = free_phys_memory_pointer;
274 
275  do{
276  /* Get current free block*/
277  tmp_block = get_free_block(cur_block);
278 
279  /* If block and count have same size*/
280  if ( tmp_block->size == count ){
281  /* Remember next and previous */
282  physaddr_t next = tmp_block->next;
283  physaddr_t prev = tmp_block->prev;
284 
285  /* Get next block */
286  tmp_block = get_free_block(next);
287 
288  /* For next block - previous is previous for allocated block */
289  tmp_block->prev = prev;
290 
291  /* Get previous block*/
292  tmp_block = get_free_block(prev);
293 
294  /* For next block - next is next for allocated block */
295  tmp_block->next = next;
296 
297  /* If it was penultimate free block */
298  if (cur_block == free_phys_memory_pointer){
299  /* Check next */
300  free_phys_memory_pointer = next;
301 
302  /* If next block is current - we have no free memory */
303  if (cur_block == free_phys_memory_pointer)
304  {
305  free_phys_memory_pointer = -1;
306  }
307  }
308 
309  /* Result address of allocated block */
310  result = cur_block;
311  break;
312  }
313 
314  /* If current free block more than count of pages */
315  if ( tmp_block->size > count ){
316  /* Reduce size of current block */
317  tmp_block->size -= count;
318 
319  /* Result address - current block's address + size of allocated block*/
320  result = cur_block + (tmp_block->size << PAGE_OFFSET_BITS);
321  break;
322  }
323 
324  /* Get next free block */
325  cur_block = tmp_block->next;
326 
327  } while (cur_block != free_phys_memory_pointer);
328 
329  /* If we have correct address */
330  if (result != -1){
331  /* Reduce free page count */
332  free_pages_count -= count;
333  }
334  }
335 
336  mutex_release(&phys_memory_mutex);
337 
338  return result;
339 }

441 
442  /* Create pages in cycle */
443  for(; count; count--){
444  uint32_t table_idx = (uint32_t) vaddr >> 22;
445  uint32_t page_idx = ((uint32_t) vaddr >> PAGE_OFFSET_BITS) & PAGE_TABLE_INDEX_MASK;
446  temp_map_page(page_dir);
