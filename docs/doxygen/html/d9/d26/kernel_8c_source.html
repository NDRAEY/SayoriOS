41  uint32_t mods_count = multiboot->mods_count;
42  char mod_cmd[16][64] = {0};
43  if (mods_count > 0){
44  mod_start = (uint32_t*) kmalloc(sizeof(uint32_t)*mods_count);
45  mod_end = (uint32_t*) kmalloc(sizeof(uint32_t)*mods_count);
46  qemu_log("[kModules] Found '%d' modules",mods_count);
47  for (size_t i = 0; i < mods_count; i++){
48  mod_start[i] = *(uint32_t*)(multiboot->mods_addr + 8*i);
49  mod_end[i] = *(uint32_t*)(multiboot->mods_addr + 8*i + 4);
50  multiboot_module_t *mod = (multiboot_module_t *) (uint32_t*)(multiboot->mods_addr + 8*i);
51  strcpy(mod_cmd[i],(char *) mod->cmdline);
52  qemu_log("[kModules] Found module number `%d`. (Start: %x | End: %x) CMD: %s",i,mod_start[i],mod_end[i],mod_cmd[i]);
53  if (strcmpn(mod_cmd[i],"initrd_sefs")){
54  initrd_sefs(mod_start[i]);
55  continue;
56  }
57  }
58  } else {
59  qemu_log("[kModules] No modules were connected to this operating system.");
60  }
