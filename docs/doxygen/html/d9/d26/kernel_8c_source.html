29 void kHandlerCMD(char* cmd){

32  uint32_t kCMDc_c = 0;
33  char* out[128] = {0};
34  str_split(cmd,out," ");

36  kCMDc_c = str_cdsp(out[i],"=");
37  char* out_data[128] = {0};
38  if (kCMDc_c != 1){
39  qemu_log("[kCMD] [%d] %s is ignore.",i,out[i]);
40  continue;
41  }
42  str_split(out[i],out_data,"=");
43  if (strcmpn(out_data[0],"bootscreen")){
44  // Config BOOTSCREEN
45  if (strcmpn(out_data[1],"minimal")){
46  bootScreenChangeMode(1);

48  bootScreenChangeTheme(1);

50  bootScreenChangeTheme(0);
51  } else {
52  qemu_log("\t Sorry, no support bootscreen mode!");
53  }
54  }
55  if (strcmpn(out_data[0],"exec")){

84 void kModules_Init(){
85  qemu_log("[kModules] Loading operating system modules...");
86  uint32_t* mod_start = 0;
87  uint32_t* mod_end = 0;
88  uint32_t mods_count = multiboot->mods_count;
89  char mod_cmd[16][64] = {0};
90  if (mods_count > 0){
91  mod_start = (uint32_t*) kmalloc(sizeof(uint32_t)*mods_count);
92  mod_end = (uint32_t*) kmalloc(sizeof(uint32_t)*mods_count);
93  qemu_log("[kModules] Found '%d' modules",mods_count);
94  for (size_t i = 0; i < mods_count; i++){
95  mod_start[i] = *(uint32_t*)(multiboot->mods_addr + 8*i);
96  mod_end[i] = *(uint32_t*)(multiboot->mods_addr + 8*i + 4);
97  multiboot_module_t *mod = (multiboot_module_t *) (uint32_t*)(multiboot->mods_addr + 8*i);
98  strcpy(mod_cmd[i],(char *) mod->cmdline);
99  qemu_log("[kModules] Found module number `%d`. (Start: %x | End: %x) CMD: %s",i,mod_start[i],mod_end[i],mod_cmd[i]);
100  if (strcmpn(mod_cmd[i],"initrd_sefs")){
101  initrd_sefs(mod_start[i]);
102  continue;
103  }
104  }
105  } else {
106  qemu_log("[kModules] No modules were connected to this operating system.");
107  }
