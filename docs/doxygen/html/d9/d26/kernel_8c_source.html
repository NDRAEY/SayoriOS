129 int get_cpu_mode() {
130  unsigned int cr0, cr4;
131 
132  __asm__ volatile("mov %%cr0, %0" : "=r"(cr0));
133  __asm__ volatile("mov %%cr4, %0" : "=r"(cr4));
134 
135  if ((cr4 & (1 << 21)) != 0) { // Bit 21 of CR4 is set, indicating long mode
136  return 64;
137  } else if ((cr0 & 1) != 0 && (cr4 & (1 << 21)) == 0) { // Bit 0 of CR0 is set and bit 21 of CR4 is clear, indicating protected mode
138  return 32;
139  } else {
140  return -1; // Unknown mode
141  }
142 }
