271 void tty_scroll() {

273  tty_pos_y -= tty_off_pos_h*num_rows;

275  // Копируем строки сверху

277  uint8_t *write_ptr = (uint8_t*) back_framebuffer_addr;

279  memcpy(write_ptr, read_ptr, num_bytes);
