127 void tty_scroll(uint32_t num_rows) {
128  uint32_t row_pos_offset = tty_off_pos_h * num_rows;
129 
130  uint8_t *addr = (uint8_t*)getFrameBufferAddr();
131  uint32_t pitch = getDisplayPitch();
132 
133  uint8_t *read_ptr = addr + (row_pos_offset * pitch);
134  uint8_t *write_ptr = addr;
135 
136  tty_pos_y -= row_pos_offset;
137 
138  uint32_t num_bytes = (pitch * VESA_HEIGHT) - (pitch * row_pos_offset);
139 
140  memcpy(write_ptr, read_ptr, num_bytes);
141 
142  // Очистка строк
143  write_ptr = addr + num_bytes;
144  memset(write_ptr, 0, pitch * row_pos_offset);
145 }
