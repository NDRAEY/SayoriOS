366 void rgba_blend(uint8_t result[4], uint8_t fg[4], uint8_t bg[4])
367 {
368  unsigned int alpha = fg[3] + 1;
369  unsigned int inv_alpha = 256 - fg[3];
370  result[0] = (uint8_t)((alpha * fg[0] + inv_alpha * bg[0]) >> 8);
371  result[1] = (uint8_t)((alpha * fg[1] + inv_alpha * bg[1]) >> 8);
372  result[2] = (uint8_t)((alpha * fg[2] + inv_alpha * bg[2]) >> 8);
373  result[3] = 0xff;

741 void _tty_print(char *format, va_list args) {
742  int32_t i = 0;
743 
744  while (format[i]) {
745  if (format[i] == '%') {
746  i++;
747  switch (format[i]) {
748  case 's':
749  _tty_puts(va_arg(args, char*));
750  break;
751  case 'c':
752  _tty_putchar(va_arg(args, int),0);
753  break;
754  case 'f':{
755  double a = va_arg(args, double);
756  if(!fpu_isInitialized()) {
757  _tty_puts("0.0000000");
758  break;
759  }
760 
761  if(a < 0) {
762  a = -a;
763  tty_puts("-");
764  }
765 
766  float rem = a-(int)a;
767  _tty_putint((int)a);
768  _tty_puts(".");
769  for(int n=0; n<7; n++) {
770  _tty_putint((int)(rem*ipow(10, n+1))%10);
771  }
772  break;
773  }
774  case 'd':
775  _tty_putint(va_arg(args, int));
776  break;
777  case 'i':
778  _tty_putint(va_arg(args, int));
779  break;
780  case 'u':
781  _tty_putint(va_arg(args, unsigned int));
782  break;
783  case 'x':
784  _tty_puthex(va_arg(args, uint32_t));
785  break;
786  case 'v':
787  _tty_puthex_v(va_arg(args, uint32_t));
788  break;
789  default:
790  _tty_putchar(format[i],format[i+1]);
791  }
792  // \n
793  } else if (format[i] == 10) {
794  tty_line_fill[tty_pos_y] = tty_pos_x;
795  tty_pos_x = 0;
796 
797  if ((tty_pos_y + tty_off_pos_h) >= (int)VESA_HEIGHT) {
798  tty_scroll();
799  } else {
800  tty_pos_y += tty_off_pos_h;
801  }
802  // \t
803  } else if (format[i] == 9) {
804  tty_pos_x += 4 * tty_off_pos_h;
805  } else {
806  _tty_putchar(format[i],format[i+1]);
807  if (isUTF(format[i])){
808  i++;
809  }
810  }
811  i++;
812  }
813 }
