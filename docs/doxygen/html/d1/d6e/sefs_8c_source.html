283 fs_node_t *sefs_initrd(uint32_t location){
284  qemu_log("[SEFS] [Init] loc: %x",location);
285  // Инициализирует указатели main и заголовке файлов и заполняет корневой директорий.
286  sefs_header = (sefs_header_t *)location;
287  file_headers = (sefs_file_header_t *) (location+sizeof(sefs_header_t));
288  sefs_root = (fs_node_t*)kmalloc(sizeof(fs_node_t));
289  strcpy(sefs_root->name, "/");
290  strcpy(sefs_root->devName, "SayoriDisk RDv2");
291  sefs_root->mask = sefs_root->uid = sefs_root->gid = sefs_root->inode = sefs_root->length = 0;
292  sefs_root->flags = FS_DIRECTORY;
293  sefs_root->open = 0;
294  sefs_root->close = 0;
295  sefs_root->findFile = &sefs_findFile;
296  sefs_root->findDir = &sefs_findDir;
297  sefs_root->getLengthFile = &sefs_getLengthFile;
298  sefs_root->getOffsetFile = &sefs_getOffsetFile;
299  sefs_root->list = &sefs_list;
300  sefs_root->ptr = 0;
301  sefs_root->impl = 0;
302  sefs_root->readChar = &sefs_readChar;
303  sefs_root->read = &sefs_read;
304  sefs_root->write = &sefs_write;
305  sefs_root->diskUsed = &sefs_diskUsed;
306  sefs_root->diskSpace = &sefs_diskSpace;
307  sefs_root->diskSize = &sefs_diskSize;
308  sefs_root->getDevName = &sefs_getDevName;
309  sefs_root->getCountElemFolder = &sefs_countElemFolder;
310  sefs_root->getListElem = &sefs_list;
311 
312  root_nodes = (fs_node_t*)kmalloc(sizeof(fs_node_t) * sefs_header->nfiles);
313  nroot_nodes = sefs_header->nfiles;
314  // Для каждого файла...
315  for (int i = 0; i < sefs_header->nfiles; i++){
316  // Отредактируем заголовок файла — в настоящее время в нем указывается смещение файла
317  // относительно ramdisk. Мы хотим, чтобы оно указывалось относительно начала
318  // памяти.
319  root_nodes[i].root = file_headers[i].parentDir;
320  file_headers[i].offset += location;
321  // Создаем нод нового файла.
322  strcpy(root_nodes[i].name, &file_headers[i].name);
323  root_nodes[i].mask = root_nodes[i].uid = root_nodes[i].gid = 0;
324  root_nodes[i].length = (uint64_t) file_headers[i].length;
325  root_nodes[i].inode = i;
326  root_nodes[i].flags = (file_headers[i].types==0?FS_FILE:FS_DIRECTORY);
327  if (root_nodes[i].flags == FS_FILE){
328  diskUsed += root_nodes[i].length;
329  diskSize += root_nodes[i].length;
330  }
331  if (root_nodes[i].flags == FS_DIRECTORY){
332  dirCount++;
333  }
334  }
335  for (int i = 0; i < sefs_header->nfiles; i++){
336  if (root_nodes[i].flags != FS_FILE) continue;
337  strcpy(root_nodes[i].path, root_nodes[(sefs_header->nfiles - dirCount) + root_nodes[i].root].name);
338  strcat(root_nodes[i].path,root_nodes[i].name);
339  int fpath_len = strlen(root_nodes[i].path);
340  root_nodes[i].path[fpath_len+1] = '\0';
341  }
342  return sefs_root;
343 }
