84 void mouse_parse_packet(const char *buf, uint8_t has_wheel, uint8_t has_5_buttons) {
85  mouse_flags_byte *mfb = (mouse_flags_byte*) (buf);
86  if (mfb->x_overflow || mfb->y_overflow || !mfb->always1) {
87  return;
88  }
89 
90  int offx = (int16_t) (0xff00 * mfb->x_sign) | buf[1];
91  int offy = (int16_t) (0xff00 * mfb->y_sign) | buf[2];
92  mouse_x += offx;
93  mouse_y -= offy;
94  mouse_b1 = mfb->left_button;
95  mouse_b2 = mfb->right_button;
96  mouse_b3 = mfb->middle_button;
97  ps2m_buffer.movement_x = offx;
98  ps2m_buffer.movement_y = offy;
99  ps2m_buffer.button_l = mouse_b1;
100  ps2m_buffer.button_r = mouse_b2;
101  ps2m_buffer.button_m = mouse_b3;
102 
103  if (mouse_b1 || mouse_b2 || mouse_b3){
104  CallTrigger(0x0003,(void*)mouse_b1,(void*)mouse_b2,(void*)mouse_b3,(void*)mouse_x,(void*)mouse_y);
105  }
106  //qemu_log("MPP: B1: %d | B2: %d | B3: %d | X:%d | Y:%d",mouse_b1,mouse_b2,mouse_b3,mouse_x,mouse_y);
107 
108  if (has_wheel) {
109  mouse_wheel += (char) ((!!(buf[3] & 0x8)) * 0xf8 | (buf[3] & 0x7));
110  if (has_5_buttons) {
111  mouse_b4 = !!(buf[3] & 0x20);
112  // parse buttons 4-5 (byte 3, bits 4-5)
113  }
114  }
115 }

166 void mouse_handler(__attribute__((unused)) struct registers r) {
167  uint8_t status = inb(0x64);
168  if ((status & 1) == 0 || (status >> 5 & 1) == 0) {
169  return;
170  }
171 
172  static int recbyte = 0;
173  static char mousebuf[5];
174 
175  mousebuf[recbyte++] = inb(0x60);
176  if (recbyte == 3 /* + has_wheel */) {
177  recbyte = 0;
178 
179  mouse_parse_packet(mousebuf, 0, 0);
180 
181  // Bounds
182  if (mouse_x < 0) mouse_x = 0;
183  if (mouse_y < 0) mouse_y = 0;
184  if (mouse_x > (int) (getScreenWidth())) mouse_x = getScreenWidth();
185  if (mouse_y > (int) (getScreenHeight())) mouse_y = getScreenHeight(); //-10;
186  CallTrigger(0x0002,(void*)mouse_x,(void*)mouse_y,0,0,0);
187  }
